1️⃣ استيراد المكون:

    import { AdvancedFilter } from '@/components/common'
    import { useDebounce } from '@/hooks'

2️⃣ إعداد الحالات:

    const [searchQuery, setSearchQuery] = useState('')
    const debouncedSearch = useDebounce(searchQuery, 500)
    const [activeFilters, setActiveFilters] = useState({})

3️⃣ تعريف الفلاتر:

    const filters = [
        {
            key: 'status',
            label: 'الحالة',
            type: 'select',
            options: [
                { value: 'active', label: 'نشط' },
                { value: 'inactive', label: 'غير نشط' }
            ]
        },
        {
            key: 'date',
            label: 'التاريخ',
            type: 'dateRange'
        }
    ]

4️⃣ جلب البيانات مع الفلاتر:

    const { data, refetch } = useQuery({
        queryKey: ['items', debouncedSearch, activeFilters],
        queryFn: () => fetch('/api/items?' + new URLSearchParams({
            search: debouncedSearch,
            ...activeFilters
        })).then(r => r.json())
    })

5️⃣ استخدام المكون:

    <AdvancedFilter
        onSearch={setSearchQuery}
        onFilterChange={setActiveFilters}
        filters={filters}
        onRefresh={refetch}
    />

📖 الخصائس (Props)

مطلوبة:
  ⚠️ onSearch: (value: string) => void
     دالة تُستدعى عند تغيير البحث

  ⚠️ onFilterChange: (filters: object) => void
     دالة تُستدعى عند تغيير أي فلتر

  ⚠️ filters: Array<FilterConfig>
     مصفوفة تعريف الفلاتر المتاحة

اختيارية:
  📌 onRefresh?: () => void
     دالة لإعادة جلب البيانات

  📌 searchPlaceholder?: string (افتراضي: 'بحث...')
     نص العنصر النائب في البحث

  📌 showResetButton?: boolean (افتراضي: true)
     إظهار/إخفاء زر إعادة التعيين

  📌 sx?: object
     أنماط Material-UI إضافية

Q: هل أحتاج لتغيير قاعدة البيانات؟
A: لا! المكون يعمل مع أي هيكل بيانات

Q: هل يدعم pagination؟
A: نعم، استخدمه مع DataTable مع pagination prop

Q: هل يمكن إضافة فلاتر مخصصة؟
A: نعم، يمكن توسيع الـ type ليدعم أنواع أخرى

Q: هل يدعم الوضع الليلي؟
A: نعم تلقائياً عبر CSS variables من الـ theme

Q: كيفية إزالة فلتر واحد فقط؟
A: انقر على X في الـ Chip الخاصة بالفلتر

Q: هل البحث فوري أم بتأخير؟
A: استخدم useDebounce للتأخير (موصى به)
